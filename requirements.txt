# =============================================================================
# Warden dependencies.
#
# The simulation engine, the CLI demo, and the web dashboard all require ZERO
# third-party packages: they run on the Python standard library alone (tested
# on 3.11, 3.12, 3.13, 3.14). Everything below is only needed to enable LIVE
# mode (real Gemini + real Dynatrace MCP + OTel push to Dynatrace).
#
# For LIVE mode on Python 3.14: google-adk does not yet target 3.14, so the
# adk_agent.py deploy path needs a 3.12 or 3.13 venv. The supervisor loop
# (google-genai + mcp + opentelemetry) runs fine on 3.14.
# =============================================================================

# LIVE mode: Warden's brain (Gemini) + MCP tool-calling
google-adk>=1.0          # canonical Agent Builder / ADK path; see warden/adk_agent.py
google-genai>=1.0        # Gemini 3 structured reasoning; warden/supervisor/gemini_brain.py
mcp>=1.2                 # MCP client to the Dynatrace MCP server; needs Node 22.10+

# LIVE mode: agent self-instrumentation shipped to Dynatrace via OTLP/HTTP.
# Dynatrace SaaS rejects gRPC; only the HTTP/protobuf exporter is supported.
opentelemetry-api>=1.42
opentelemetry-sdk>=1.42
opentelemetry-exporter-otlp-proto-http>=1.42
